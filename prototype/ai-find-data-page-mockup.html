-- AI 根据你的需求生成：查看最近7天各渠道的退款金额
SELECT
  refund_channel,
  DATE(refund_time) AS refund_date,
  COUNT(*) AS refund_cnt,
  SUM(refund_amount) AS total_refund_amt
FROM dwd_trade_refund_detail
WHERE refund_time >= DATE_SUB(CURDATE(), INTERVAL 7 DAY)
GROUP BY refund_channel, DATE(refund_time)
ORDER BY refund_date DESC, total_refund_amt DESC;
